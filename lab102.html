let sortbylastname = function() { let tbl = document.getElementsByTagName('table')[0]; let rows = Array.from(tbl.rows); // Convert NodeList to array // Clear existing rows tbl.innerHTML = ''; // Sort rows by first name (assuming first column is first name) rows.sort((a, b) => { return a.cells[0].innerHTML.localeCompare(b.cells[0].innerHTML); }); // Append sorted rows back to table rows.forEach(row => { tbl.appendChild(row); }); };
