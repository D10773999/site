function sortbyfirstname() { let tbl = document.getElementsByTagName('table')[0]; let rows = []; for (let i = 0; i < tbl.rows.length; i++) { rows.push(tbl.rows[i]); } rows.sort((a, b) => { const aFirst = a.cells[0].innerHTML.toLowerCase(); const bFirst = b.cells[0].innerHTML.toLowerCase(); return aFirst.localeCompare(bFirst); }); while (tbl.firstChild) { tbl.removeChild(tbl.firstChild); } rows.forEach(row => tbl.appendChild(row)); } window.onload = sortbyfirstname;
